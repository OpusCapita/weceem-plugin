001package com.jcatalog.wiki.block;
002
003import net.java.textilej.parser.Attributes;
004import net.java.textilej.parser.DocumentBuilder.BlockType;
005import net.java.textilej.parser.markup.Block;
006import net.java.textilej.parser.markup.confluence.ConfluenceDialect;
007
008/**
009 * Class is the same as net.java.textilej.parser.markup.confluence.block.ParagraphBlock,
010 * except 'confluenceBlockStart' constant value, because processing for
011 * {code}, {info}, {tip}, {warning}, {panel}, {note} and {toc} blocks don't needed.
012 */
013public class ParagraphBlock extends Block {
014
015    private int blockLineCount = 0;
016
017    /**
018     * @see net.java.textilej.parser.markup.Block#processLineContent(java.lang.String, int)
019     */
020    @Override
021    public int processLineContent(String line, int offset) {
022        if (blockLineCount == 0) {
023            Attributes attributes = new Attributes();
024
025            builder.beginBlock(BlockType.PARAGRAPH, attributes);
026        }
027
028        if (dialect.isEmptyLine(line)) {
029            setClosed(true);
030            return 0;
031        }
032
033        ++blockLineCount;
034
035        ConfluenceDialect myDialect = (ConfluenceDialect) getDialect();
036
037        // NOTE: in Textile paragraphs can have nested lists and other things, however
038        //       the resulting XHTML is invalid -- so here we allow for similar constructs
039        //       however we cause them to end the paragraph rather than being nested.
040        for (Block block : myDialect.getParagraphBreakingBlocks()) {
041            if (block.canStart(line, offset)) {
042                setClosed(true);
043                return 0;
044            }
045        }
046
047        if (blockLineCount > 1) {
048            builder.lineBreak();
049        }
050        myDialect.emitMarkupLine(getParser(), state, line, offset);
051
052        return -1;
053    }
054
055    /**
056     * @see net.java.textilej.parser.markup.Block#canStart(java.lang.String, int)
057     */
058    @Override
059    public boolean canStart(String line, int lineOffset) {
060        blockLineCount = 0;
061        return true;
062    }
063
064    /**
065     * @see net.java.textilej.parser.markup.Block#setClosed(boolean)
066     */
067    @Override
068    public void setClosed(boolean closed) {
069        if (closed && !isClosed()) {
070            builder.endBlock();
071        }
072        super.setClosed(closed);
073    }
074}
