001package com.jcatalog.wiki;
002
003import com.jcatalog.wiki.block.ParagraphBlock;
004import net.java.textilej.parser.markup.Block;
005import net.java.textilej.parser.markup.confluence.ConfluenceDialect;
006import net.java.textilej.parser.markup.confluence.block.CodeBlock;
007import net.java.textilej.parser.markup.confluence.block.ExtendedPreformattedBlock;
008import net.java.textilej.parser.markup.confluence.block.ExtendedQuoteBlock;
009import net.java.textilej.parser.markup.confluence.block.HeadingBlock;
010import net.java.textilej.parser.markup.confluence.block.ListBlock;
011import net.java.textilej.parser.markup.confluence.block.QuoteBlock;
012import net.java.textilej.parser.markup.confluence.block.TableBlock;
013
014import java.util.ArrayList;
015import java.util.List;
016
017/**
018 * The same markup dialect as Confluence dialect except it doesn't process
019 * macros such as {panel}, {note}, {warning}, {info}, {tip}, {toc}.
020 */
021public class WeceemDialect extends ConfluenceDialect {
022
023    private List<Block> supportedBlocks;
024
025    /**
026     * 
027     */
028    public WeceemDialect() {
029        supportedBlocks = new ArrayList<Block>();
030        supportedBlocks.add(new HeadingBlock());
031        supportedBlocks.add(new ListBlock());
032        supportedBlocks.add(new QuoteBlock());
033        supportedBlocks.add(new TableBlock());
034        supportedBlocks.add(new ExtendedQuoteBlock());
035        supportedBlocks.add(new ExtendedPreformattedBlock());
036        supportedBlocks.add(new CodeBlock());
037        supportedBlocks.add(new ParagraphBlock());
038    }
039
040    /**
041     * @see net.java.textilej.parser.markup.confluence.ConfluenceDialect#getBlocks()
042     */
043    @Override
044    public List<Block> getBlocks() {
045        return supportedBlocks;
046    }
047}
